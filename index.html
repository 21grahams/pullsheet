const SINGLES_TAB = 'Market Trends Singles'; const SEALED_TAB = 'Market Trends Sealed'; // ── CORS wrapper ──────────────────────────────────────────── function doGet(e) { let result; try { if (e.parameter.payload) { const payload = JSON.parse(decodeURIComponent(e.parameter.payload)); const action = payload.action; if (action === 'addSingle') result = addSingle(payload.data); else if (action === 'updateSingleMarket') result = updateSingleMarket(payload.data); else if (action === 'deleteSingle') result = deleteSingle(payload.data); else if (action === 'addSealed') result = addSealed(payload.data); else if (action === 'updateSealedMarket') result = updateSealedMarket(payload.data); else if (action === 'deleteSealed') result = deleteSealed(payload.data); else if (action === 'markSold') result = markSold(payload.data); else result = { error: 'Unknown action: ' + action }; } else { const action = e.parameter.action; if (action === 'getSingles') result = getSingles(); else if (action === 'getSealed') result = getSealed(); else if (action === 'getSummary') result = getSummary(); else result = { error: 'Unknown action: ' + action }; } } catch(err) { result = { error: err.message }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { let result; try { const payload = JSON.parse(e.postData.contents); const action = payload.action; if (action === 'addSingle') result = addSingle(payload.data); else if (action === 'updateSingleMarket') result = updateSingleMarket(payload.data); else if (action === 'deleteSingle') result = deleteSingle(payload.data); else if (action === 'addSealed') result = addSealed(payload.data); else if (action === 'updateSealedMarket') result = updateSealedMarket(payload.data); else if (action === 'deleteSealed') result = deleteSealed(payload.data); else if (action === 'markSold') result = markSold(payload.data); else result = { error: 'Unknown action' }; } catch(err) { result = { error: err.message }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ── READ ──────────────────────────────────────────────────── function getSingles() { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SINGLES_TAB); const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = []; for (let i = 1; i < data.length; i++) { const row = data[i]; if (!row[0]) continue; // skip empty rows rows.push({ rowIndex: i + 1, card: row[0], quantity: row[1], purchaseDate: row[2] ? Utilities.formatDate(new Date(row[2]), Session.getScriptTimeZone(), 'MM/dd/yyyy') : '', purchasePrice: row[3], currentMarket: row[4], diff: row[5], pctChange: row[6], }); } return { singles: rows }; } function getSealed() { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SEALED_TAB); const data = sheet.getDataRange().getValues(); const rows = []; let currentGroup = 'Long Hold'; for (let i = 1; i < data.length; i++) { const row = data[i]; if (!row[0]) continue; // detect group header rows if (typeof row[0] === 'string' && row[0].trim() !== '' && !row[1] && !row[3]) { currentGroup = row[0].trim(); continue; } if (!row[3]) continue; // skip rows without purchase price rows.push({ rowIndex: i + 1, product: row[0], quantity: row[1], purchaseDate: row[2] ? Utilities.formatDate(new Date(row[2]), Session.getScriptTimeZone(), 'MM/dd/yyyy') : '', purchasePrice: row[3], currentMarket: row[4], diff: row[5], pctChange: row[6], group: currentGroup, soldPrice: row[7] || null, profit: row[8] || null, }); } return { sealed: rows }; } function getSummary() { const singles = getSingles().singles; const sealed = getSealed().sealed; const singleSpent = singles.reduce((s, r) => s + (parseFloat(r.purchasePrice) || 0) * (parseInt(r.quantity) || 1), 0); const singleValue = singles.reduce((s, r) => s + (parseFloat(r.currentMarket) || 0) * (parseInt(r.quantity) || 1), 0); const longHold = sealed.filter(r => r.group === 'Long Hold'); const shortHold = sealed.filter(r => r.group !== 'Long Hold'); const longSpent = longHold.reduce((s, r) => s + (parseFloat(r.purchasePrice) || 0) * (parseInt(r.quantity) || 1), 0); const longValue = longHold.reduce((s, r) => s + (parseFloat(r.currentMarket) || 0) * (parseInt(r.quantity) || 1), 0); const shortSpent = shortHold.reduce((s, r) => s + (parseFloat(r.purchasePrice) || 0) * (parseInt(r.quantity) || 1), 0); const shortValue = shortHold.reduce((s, r) => s + (parseFloat(r.currentMarket) || 0) * (parseInt(r.quantity) || 1), 0); const grandSpent = singleSpent + longSpent + shortSpent; const grandValue = singleValue + longValue + shortValue; return { singles: { spent: singleSpent, value: singleValue, diff: singleValue - singleSpent }, longHold: { spent: longSpent, value: longValue, diff: longValue - longSpent, market80: longValue * 0.8 }, shortHold: { spent: shortSpent, value: shortValue, diff: shortValue - shortSpent, market80: shortValue * 0.8 }, grand: { spent: grandSpent, value: grandValue, diff: grandValue - grandSpent, market80: grandValue * 0.8 } }; } // ── WRITE ─────────────────────────────────────────────────── function addSingle(data) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SINGLES_TAB); const market = parseFloat(data.currentMarket) || 0; const cost = parseFloat(data.purchasePrice) || 0; const qty = parseInt(data.quantity) || 1; const diff = (market - cost) * qty; const pct = cost > 0 ? ((market - cost) / cost) * 100 : 0; sheet.appendRow([ data.card, qty, new Date(data.purchaseDate), cost, market, diff, pct / 100, ]); return { success: true }; } function updateSingleMarket(data) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SINGLES_TAB); const row = parseInt(data.rowIndex); const market = parseFloat(data.currentMarket); const cost = parseFloat(sheet.getRange(row, 4).getValue()); const qty = parseInt(sheet.getRange(row, 2).getValue()) || 1; sheet.getRange(row, 5).setValue(market); sheet.getRange(row, 6).setValue((market - cost) * qty); sheet.getRange(row, 7).setValue(cost > 0 ? ((market - cost) / cost) : 0); return { success: true }; } function deleteSingle(data) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SINGLES_TAB); sheet.deleteRow(parseInt(data.rowIndex)); return { success: true }; } function addSealed(data) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SEALED_TAB); const market = parseFloat(data.currentMarket) || 0; const cost = parseFloat(data.purchasePrice) || 0; const qty = parseInt(data.quantity) || 1; const diff = (market - cost) * qty; const pct = cost > 0 ? ((market - cost) / cost) : 0; // Find the right section to insert into const allData = sheet.getDataRange().getValues(); let insertRow = sheet.getLastRow() + 1; const targetGroup = data.holdType === 'long' ? 'Long Hold' : data.soldPeriod || 'Short Hold One'; for (let i = 0; i < allData.length; i++) { if (typeof allData[i][0] === 'string' && allData[i][0].trim() === targetGroup) { // find end of this group for (let j = i + 1; j < allData.length; j++) { if (typeof allData[j][0] === 'string' && allData[j][0].trim() !== '' && !allData[j][3]) { insertRow = j + 1; break; } if (j === allData.length - 1) insertRow = j + 2; } break; } } sheet.insertRowBefore(insertRow); sheet.getRange(insertRow, 1, 1, 7).setValues([[ data.product, qty, new Date(data.purchaseDate), cost, market, diff, pct ]]); return { success: true }; } function updateSealedMarket(data) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SEALED_TAB); const row = parseInt(data.rowIndex); const market = parseFloat(data.currentMarket); const cost = parseFloat(sheet.getRange(row, 4).getValue()); const qty = parseInt(sheet.getRange(row, 2).getValue()) || 1; sheet.getRange(row, 5).setValue(market); sheet.getRange(row, 6).setValue((market - cost) * qty); sheet.getRange(row, 7).setValue(cost > 0 ? ((market - cost) / cost) : 0); return { success: true }; } function deleteSealed(data) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SEALED_TAB); sheet.deleteRow(parseInt(data.rowIndex)); return { success: true }; } function markSold(data) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SEALED_TAB); const row = parseInt(data.rowIndex); sheet.getRange(row, 8).setValue(parseFloat(data.soldPrice)); const cost = parseFloat(sheet.getRange(row, 4).getValue()); const qty = parseInt(sheet.getRange(row, 2).getValue()) || 1; sheet.getRange(row, 9).setValue(parseFloat(data.soldPrice) - cost * qty); return { success: true }; }
